# Create a vector of numbers
numbers <- c(10, 20, 30, 40, 50, 60)

# Create a matrix using the vector
mat <- matrix(numbers, nrow = 2, ncol = 3, byrow = TRUE)

# Define row and column names
rownames(mat) <- c("Row1", "Row2")
colnames(mat) <- c("Col1", "Col2", "Col3")

# Display the matrix
print("Matrix with Row and Column Names:")
print(mat)



# Create the first 2x3 matrix
matrix1 <- matrix(c(2, 4, 6, 8, 10, 12), nrow = 2, ncol = 3)

# Create the second 2x3 matrix
matrix2 <- matrix(c(1, 3, 5, 7, 9, 11), nrow = 2, ncol = 3)

# Display both matrices
print("First Matrix:")
print(matrix1)

print("Second Matrix:")
print(matrix2)

# Addition of matrices
add_result <- matrix1 + matrix2
print("Addition of Matrices:")
print(add_result)

# Subtraction of matrices
sub_result <- matrix1 - matrix2
print("Subtraction of Matrices:")
print(sub_result)

# Multiplication of matrices (element-wise)
mul_result <- matrix1 * matrix2
print("Multiplication of Matrices:")
print(mul_result)

# Division of matrices (element-wise)
div_result <- matrix1 / matrix2
print("Division of Matrices:")
print(div_result)



# Create a list of vectors
list_vectors <- list(
  c(1, 2, 3),
  c(4, 5, 6),
  c(7, 8, 9)
)

# Combine the list elements into a matrix
mat <- do.call(rbind, list_vectors)

# Display the matrix
print("Matrix created from a list of vectors:")
print(mat)




# Create a matrix
mat <- matrix(1:9, nrow = 3, ncol = 3)
print("Original Matrix:")
print(mat)

# Convert the matrix to a list of column vectors
list_columns <- split(mat, col(mat))

# Display the list
print("List of Column Vectors:")
print(list_columns)
